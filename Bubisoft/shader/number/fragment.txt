#version 330 core
    in vec2 bTex;
    in vec3 bColor;
    in float bAlpha;
    uniform sampler2D textureSampler;
    out vec4 FragColor;
    void main()
    {
        vec4 pc = vec4(texture(textureSampler, bTex));
	float sum = pc.x + pc.y + pc.z; 
	if( sum < 0.15 ) { discard; }
	else { 
          pc.w = bAlpha;
          FragColor = pc * vec4(bColor.xyz, 1.0f);
        }
    }
